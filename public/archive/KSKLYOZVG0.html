devinfd commented Nov 9, 2015

As a follow up. I created a trait SudoDatabaseTransactions that stopped the cleanup/transaction functionality but still "cleaned" the written data.

Basically this trait has a _before method that counteracts the laravel5 module _before method and an _after method that truncates the tables. And to get this behavior I just use the trait in my test class.

namespace Helper;

trait SudoDatabaseTransactions
{
    /**
     * Database tables to clean
     *
     * @var array
     */
    protected $tables = [
        'users',
        'posts',
        'comments'
    ];

    /**
     * @before
     */
    public function _beginSudoDatabaseTransaction()
    {
        $this->getModule('Laravel5')->getApplication()['db']->rollback();

        $this->getModule('Laravel5')->config['cleanup'] = false;
    }

    /**
     * @after
     */
    public function _endSudoDatabaseTransaction()
    {
        $laravel = $this->getModule('Laravel5');
        $db = $laravel->getApplication()['db'];

        $db->statement('SET FOREIGN_KEY_CHECKS=0');

        foreach ($this->tables as $table) {
            $db->table($table)->truncate();
        }

        $db->statement('SET FOREIGN_KEY_CHECKS=1');

        $laravel->config['cleanup'] = true;
    }
}
